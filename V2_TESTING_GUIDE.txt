cursor.execute('SELECT budget_id, category, amount, spent_amount, status FROM budgets LIMIT 5')

print('\\n=== Alerts ===')
cursor.execute('SELECT alert_id, title, severity, is_read FROM alerts LIMIT 5')
for row in cursor.fetchall():
    read_status = 'READ' if row[3] else 'UNREAD'
    print(f'Alert #{row[0]}: {row[1]} - Severity: {row[2]} - {read_status}')

conn.close()

Goals:
  POST   /api/v1/goals/                          Create goal
  GET    /api/v1/goals/{user_id}                 List all goals
  GET    /api/v1/goals/{user_id}/{goal_id}       Get specific goal
  PUT    /api/v1/goals/{user_id}/{goal_id}       Update goal
  DELETE /api/v1/goals/{user_id}/{goal_id}       Delete goal
  POST   /api/v1/goals/{user_id}/{goal_id}/progress  Recalculate progress

Budgets:
  POST   /api/v1/budgets/                        Create budget
  GET    /api/v1/budgets/{user_id}               List all budgets
  GET    /api/v1/budgets/{user_id}/summary       Get budget summary
  GET    /api/v1/budgets/{user_id}/{budget_id}   Get specific budget
  PUT    /api/v1/budgets/{user_id}/{budget_id}   Update budget
  DELETE /api/v1/budgets/{user_id}/{budget_id}   Delete budget
  POST   /api/v1/budgets/{user_id}/auto-generate Auto-generate budgets
  POST   /api/v1/budgets/{user_id}/{budget_id}/refresh  Refresh spending

Alerts:
  POST   /api/v1/alerts/                         Create alert
  GET    /api/v1/alerts/{user_id}                List alerts (with filters)
  GET    /api/v1/alerts/{user_id}/unread-count   Get unread count
  GET    /api/v1/alerts/{user_id}/{alert_id}     Get specific alert
  PUT    /api/v1/alerts/{user_id}/{alert_id}     Update alert (read/dismiss)
  POST   /api/v1/alerts/{user_id}/mark-all-read  Mark all as read
  DELETE /api/v1/alerts/{user_id}/{alert_id}     Delete alert
  POST   /api/v1/alerts/{user_id}/generate       Auto-generate alerts
